for(单次表达式;条件表达式;末尾循环体)
{
中间循环体;
}

#include
using namespace std;
int main()
{
for(int i = 0;i <10;i++)
{
cout <<”我吃了”<<i<<”颗糖”<<endl;
}
return 0;
}

#include
using namespace std;
int main()
{
int MAX=5;
int a[MAX]={10,20,30,40,50};
for(int i = 0;i<MAX;i++)
{
cout << a[i] <<endl;
}
return 0;
}
